# Twin   backend dependencies.
# Versions are pinned exactly. LangChain's APIs move fast; pinning is what stops a
# `pip install` on the morning of the demo from breaking a working app.
#
# Verified on Python 3.11–3.13 (Windows/macOS/Linux).

# --- Web -------------------------------------------------------------------------------
Flask==3.1.0
Flask-Cors==5.0.0
requests==2.32.3
python-dotenv==1.0.1
gunicorn==23.0.0

# --- RAG: LangChain + ChromaDB ---------------------------------------------------------
# chromadb 1.x ships prebuilt wheels for Python 3.13. Older 0.5.x pulls chroma-hnswlib,
# which has no 3.13 wheel and tries to compile from source.
chromadb==1.5.9
langchain-core==1.4.9
langchain-chroma==1.1.0
langchain-text-splitters==1.1.2

# --- Embeddings ------------------------------------------------------------------------
# Tier 1: OpenAI text-embedding-3-small, used only if OPENAI_API_KEY is set.
langchain-openai==1.3.5
# Tier 2 (local sentence-transformers/all-MiniLM-L6-v2) is deliberately NOT installed.
# It pulls torch and downloads the model on first use, which takes longer than gunicorn's
# 30s worker timeout on a cold start   that is what made /api/health and /api/chat return
# 500 in production while /api/profile and /api/timeline stayed fine.
# Twin falls back to the built-in hashing embeddings in rag.py: no download, no torch, and
# it still ingests and retrieves over the knowledge base.
# To use it locally anyway:
#   pip install langchain-huggingface==1.2.2 sentence-transformers==3.3.1

# --- Tests -----------------------------------------------------------------------------
pytest==8.3.4
